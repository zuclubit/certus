import bpy
import math

# === SCENE SETUP ===
scene = bpy.context.scene

# Render engine
scene.render.engine = 'CYCLES'
scene.cycles.device = 'GPU'  # Use GPU if available
scene.cycles.samples = 2048  # Ultra-high quality
scene.cycles.use_adaptive_sampling = True
scene.cycles.adaptive_threshold = 0.01
scene.cycles.use_denoising = True
scene.cycles.denoiser = 'OPENIMAGEDENOISE'

# Resolution (4K square)
scene.render.resolution_x = 4096
scene.render.resolution_y = 4096
scene.render.resolution_percentage = 100
scene.render.film_transparent = True  # Alpha channel

# Color management (ACES)
scene.view_settings.view_transform = 'AgX'  # or ACES if available
scene.view_settings.look = 'Medium High Contrast'
scene.display_settings.display_device = 'sRGB'

# Light paths
scene.cycles.max_bounces = 12
scene.cycles.diffuse_bounces = 4
scene.cycles.glossy_bounces = 8
scene.cycles.transmission_bounces = 12
scene.cycles.volume_bounces = 2
scene.cycles.transparent_max_bounces = 16

# Caustics
scene.cycles.caustics_reflective = True
scene.cycles.caustics_refractive = True
scene.cycles.blur_glossy = 0.5

# === CAMERA SETUP ===
camera_data = bpy.data.cameras.new(name="Icon_Camera")
camera_data.type = 'ORTHO'
camera_data.ortho_scale = 2.8
camera_data.clip_start = 0.1
camera_data.clip_end = 100

camera_obj = bpy.data.objects.new("Icon_Camera", camera_data)
bpy.context.collection.objects.link(camera_obj)
scene.camera = camera_obj

# Position (front view with slight tilt)
camera_obj.location = (0, -12, 0.8)
camera_obj.rotation_euler = (math.radians(86), 0, 0)

# === HEXAGON GEOMETRY ===
bpy.ops.mesh.primitive_cylinder_add(
    vertices=6,
    radius=1.0,
    depth=0.08,
    location=(0, 0, 0)
)
hexagon = bpy.context.active_object
hexagon.name = "Hexagon_Glass"

# Smooth shading
bpy.ops.object.shade_smooth()

# Add bevel modifier for rounded edges
bevel = hexagon.modifiers.new(name="Bevel", type='BEVEL')
bevel.width = 0.02
bevel.segments = 4
bevel.limit_method = 'ANGLE'

# Add subdivision for smoothness
subsurf = hexagon.modifiers.new(name="Subdivision", type='SUBSURF')
subsurf.levels = 2
subsurf.render_levels = 3

# === GLASS MATERIAL ===
glass_mat = bpy.data.materials.new(name="Liquid_Glass")
glass_mat.use_nodes = True
nodes = glass_mat.node_tree.nodes
links = glass_mat.node_tree.links

# Clear default nodes
nodes.clear()

# Add Principled BSDF
bsdf = nodes.new(type='ShaderNodeBsdfPrincipled')
bsdf.location = (0, 0)

# Glass properties
bsdf.inputs['Base Color'].default_value = (1, 1, 1, 1)  # White
bsdf.inputs['Metallic'].default_value = 0.0
bsdf.inputs['Roughness'].default_value = 0.05
bsdf.inputs['IOR'].default_value = 1.45
bsdf.inputs['Transmission'].default_value = 0.88
bsdf.inputs['Transmission Roughness'].default_value = 0.05

# Add output
output = nodes.new(type='ShaderNodeOutputMaterial')
output.location = (200, 0)
links.new(bsdf.outputs['BSDF'], output.inputs['Surface'])

# Assign material
hexagon.data.materials.append(glass_mat)

# === LIGHTING SETUP ===

# Key Light (Area)
bpy.ops.object.light_add(type='AREA', location=(-5, 8, 5))
key_light = bpy.context.active_object
key_light.name = "Key_Light"
key_light.data.energy = 250
key_light.data.color = (0.973, 0.980, 0.988)  # #F8FAFC
key_light.data.size = 2.0
key_light.rotation_euler = (math.radians(45), 0, math.radians(-45))

# Rim Light (Spot)
bpy.ops.object.light_add(type='SPOT', location=(8, 3, -3))
rim_light = bpy.context.active_object
rim_light.name = "Rim_Light"
rim_light.data.energy = 150
rim_light.data.color = (0.376, 0.647, 0.980)  # #60A5FA
rim_light.data.spot_size = math.radians(45)
rim_light.data.spot_blend = 0.15

# HDRI Environment
world = bpy.context.scene.world
world.use_nodes = True
world_nodes = world.node_tree.nodes
world_links = world.node_tree.links

# Clear default
world_nodes.clear()

# Environment Texture
env_tex = world_nodes.new(type='ShaderNodeTexEnvironment')
env_tex.location = (-300, 300)
# Load HDRI here: env_tex.image = bpy.data.images.load("path/to/studio.hdr")

# Background shader
background = world_nodes.new(type='ShaderNodeBackground')
background.location = (0, 300)
background.inputs['Strength'].default_value = 1.2

# Output
world_output = world_nodes.new(type='ShaderNodeOutputWorld')
world_output.location = (200, 300)

world_links.new(env_tex.outputs['Color'], background.inputs['Color'])
world_links.new(background.outputs['Background'], world_output.inputs['Surface'])

print("Certus V3 scene setup complete!")
